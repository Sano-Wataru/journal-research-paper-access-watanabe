\revised{For normals (Fig.~\ref{valacc}\subref{valnormal}), validation accuracy remained high under all clustering settings.}
  %2-1
  \revised{Normal features assign one direction per point and are less sensitive to point distribution changes caused by clustering.}
%3
\revised{For dimensionality features (Fig.~\ref{valacc}\subref{valdim}), validation accuracy remained high; k-means showed a slight decrease relative to no clustering and DBSCAN.}
  %3-1
  \revised{k-means can remove points that carry orientation information needed for dimensionality features.}
%4
\revised{For proportion of variance (Fig.~\ref{valacc}\subref{valproportion}), validation accuracy remained high with a similar k-means decrease.}
%5
\revised{For FPFH (Fig.~\ref{valacc}\subref{valfpfh}), validation accuracy remained high; DBSCAN showed larger epoch-to-epoch variation than the other clustering settings.}
  %5-1
  \revised{DBSCAN varies cluster counts across frames and alters the local shape encoded by FPFH.}
