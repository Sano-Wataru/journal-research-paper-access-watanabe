% Revision markup: deleted = red, revised = blue
\newcommand{\revised}[1]{{\color[rgb]{0,0,1}#1}}
\newcommand{\deleted}[1]{{\color[rgb]{1,0,0}#1}}

% Block markup: color only (strikethrough is not compatible with floats/citations)
\NewEnviron{revisedblock}{%
  \begingroup
  \color[rgb]{0,0,1}%
  \BODY
  \endgroup
}
\NewEnviron{deletedblock}{%
  \begingroup
  \color[rgb]{1,0,0}%
  \BODY
  \endgroup
}
